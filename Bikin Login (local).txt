    public void logins(){
        String username=txtUser.getText();
        String password=txtPass.getText();
        String sql;

        
        try {
            Class.forName("com.mysql.jdbc.Driver");
            conn = (Connection) DriverManager.getConnection("jdbc:mysql://"+host+"/"+db+"?user="+user+"&password="+pass);
            sql ="Select * from user WHERE username ='" +txtUser.getText() + "'";
            PreparedStatement st = (PreparedStatement) conn.prepareStatement(sql);
            ResultSet rs =st.executeQuery();      
            if(rs.next()) {
                String pasas = rs.getString("password");
                if(password.equals(pasas)){
                    JOptionPane.showMessageDialog(this, "Login Berhasil");
                    menu_utama tampil = new menu_utama();
                    tampil.setVisible(true);
                    dispose();
                }else{
                    JOptionPane.showMessageDialog(this, "Username tidak terdaftar");
                    txtUser.requestFocus();
                }
            }else{
                JOptionPane.showMessageDialog(this, "Username tidak terdaftar");
                txtUser.requestFocus();
            }
        }catch (Exception e) {
            JOptionPane.showMessageDialog(this,e.getMessage());
        }
        
    }//akhir method login();
